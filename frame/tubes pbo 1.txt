id_admin varchar (10),
nama_pegawai varchar (30),
jenis_kelamin varchar (10),
jabatan varchar (30),
primary key (id_pegawai);

Produk :
produk_id varchar (10),
nama_produk varchar (30),
jumlah_stok varchar (10),
Harga_jual varchar (30),
primary key(produk_id);

Pemasok :
id_pemasok varchar (10),
nama varchar (30),
No_telp varchar (20),
primary key (id_pemasok);

Pembeli :
id_pembeli varchar (10),
nama_pembeli varchar(30),
No_telp varchar (20),
jumlah_beli varchar (20),\s
primary key(id-pembeli);

Transaksi :
id_transaksi varchar (10),
id_pembeli varchar (10),
